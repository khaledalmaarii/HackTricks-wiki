import requests

url = "http://example.com:80/some/path.php"

headers = {"User-Agent": "fake User Agent", "Fake header": "True value"}

proxies = {'http':'http://127.0.0.1:8080','https':'http://127.0.0.1:8080'}

#Regular Get requests sending parameters (params)
